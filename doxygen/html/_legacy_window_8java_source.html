109 public static void tone(int hz, int msecs, double vol)
110 throws LineUnavailableException
111 {
112 byte[] buf = new byte[1];
113 AudioFormat af =
114 new AudioFormat(
115 SAMPLE_RATE, // sampleRate
116 8, // sampleSizeInBits
117 1, // channels
118 true, // signed
119 false); // bigEndian
120 SourceDataLine sdl = AudioSystem.getSourceDataLine(af);
121 sdl.open(af);
122 sdl.start();
123 for (int i=0; i < msecs*8; i++) {
124 double angle = i / (SAMPLE_RATE / hz) * 2.0 * Math.PI;
125 buf[0] = (byte)(Math.sin(angle) * 127.0 * vol);
126 sdl.write(buf,0,1);
127 }
128 sdl.drain();
129 sdl.stop();
130 sdl.close();
131 }
