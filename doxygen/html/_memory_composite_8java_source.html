350 private String[] search(String[] entries, String toSearch)
351 {
352 ArrayList<String> filtered = new ArrayList<String>();
353 for(int index = 0; index < entries.length; index++)
354 {
355 if(toSearch.endsWith("*"))
356 {
357 if(entries[index].toLowerCase().substring(entries[index].indexOf('}')+1).startsWith(toSearch.toLowerCase().substring(0, toSearch.length()-1)))
358 {
359 filtered.add(entries[index]);
360 }
361 }
362 else if(toSearch.startsWith("*"))
363 {
364 if(entries[index].toLowerCase().substring(entries[index].indexOf('}')+1).endsWith(toSearch.toLowerCase().substring(1)))
365 {
366 filtered.add(entries[index]);
367 }
368 }
369 else
370 {
371 if(entries[index].toLowerCase().substring(entries[index].indexOf('}')+1).contains(toSearch.toLowerCase()))
372 {
373 filtered.add(entries[index]);
374 }
375 }
376
377 }
378 String[] filteredArray = filtered.toArray(new String[0]);
