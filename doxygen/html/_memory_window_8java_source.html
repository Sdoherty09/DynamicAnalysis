311 private String[] search(String[] entries, String toSearch)
312 {
313 ArrayList<String> filtered = new ArrayList<String>();
314 for(int index = 0; index < entries.length; index++)
315 {
316 if(toSearch.endsWith("*"))
317 {
318 if(entries[index].toLowerCase().startsWith(toSearch.toLowerCase().substring(0, toSearch.length()-1)))
319 {
320 filtered.add(entries[index]);
321 }
322 }
323 else if(toSearch.startsWith("*"))
324 {
325 if(entries[index].toLowerCase().endsWith(toSearch.toLowerCase().substring(1)))
326 {
327 filtered.add(entries[index]);
328 }
329 }
330 else
331 {
332 if(entries[index].toLowerCase().contains(toSearch.toLowerCase()))
333 {
334 filtered.add(entries[index]);
335 }
336 }
337
338 }
339 String[] filteredArray = filtered.toArray(new String[0]);
340 return filteredArray;
341 }
