305 private synchronized void updateNetwork()
306 {
307 Thread updateThread = new Thread(() -> { // Set up thread for each network interface
308 while (true)
309 {
310 HashMap<String, String> newDevices = packetTrace.getDevices();
311
312 Display.getDefault().asyncExec(new Runnable() {
313 public void run() {
314 if(!newDevices.equals(devices))
315 {
316 devices = newDevices;
317 networkInterfaces.removeAll();
318 for(Entry<String, String> device : devices.entrySet())
319 {
320 networkInterfaces.add(device.getValue());
321 networkInterfaces.setData(device.getValue(), device.getKey());
322 }
323 }
324 if(addressSelected)
325 {
326 try
327 {
328 ArrayList<IpPacket> packetList = packetTrace.getPackets(addresses.getSelection()[0], 0);
329 if(packetList == null)
330 {
331 errorAlert("Something went wrong when fetching packets.");
332 }
333 for(IpPacket packet : packetList)
334 {
335 String packetDescription = packet.getHeader().getProtocol().name()+" - "+packet.getPayload().getRawData().length+" bytes {"+packet.hashCode()+"}";
336 packets.add(packetDescription);
337 packets.setData(packetDescription, packet);
338 }
339 }
340 catch(ArrayIndexOutOfBoundsException e) {}
341 }
342 }
343 });
344 try
345 {
346 TimeUnit.SECONDS.sleep(1);
347 } catch (InterruptedException e)
348 {
349 e.printStackTrace();
350 }
351 }
352 });
353 updateThread.start();
354 }
