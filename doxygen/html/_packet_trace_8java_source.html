56 public PacketTrace() throws PcapNativeException
57 {
58 devices = Pcaps.findAllDevs();
59 for(int index = 0;index < devices.size();index++)
60 {
61 idMap.put(devices.get(index).getName(), devices.get(index).getDescription());
62 interfaceMap.put(devices.get(index).getName(), index); // Separating from multithread to avoid synchronization issues
63 packets.add(ArrayListMultimap.create());
64 }
65 for (PcapNetworkInterface device : devices)
66 {
67 PcapHandle handle = device.openLive(65536, PcapNetworkInterface.PromiscuousMode.PROMISCUOUS, 10);
68 try {
69 handle.setFilter("ip", BpfProgram.BpfCompileMode.OPTIMIZE);
70 } catch (NotOpenException e1) {
71 e1.printStackTrace();
72 }
73 listenerThread = new Thread(() -> { // Set up thread for each network interface
74 try {
75 while (true) {
76 Packet packet = handle.getNextPacket();
77 if(packet!=null)
78 {
79 if (packet.contains(IpPacket.class)) {
80 IpPacket ipPacket = packet.get(IpPacket.class);
81 if(!addressMap.containsKey(ipPacket.getHeader().getSrcAddr().getHostAddress())) addressMap.put(device.getName(), ipPacket.getHeader().getSrcAddr().getHostAddress());
82 packets.get(interfaceMap.get(device.getName())).put(ipPacket.getHeader().getSrcAddr().getHostAddress(), ipPacket); //needs address
83 }
84 }
85
86 }
87 } catch (NotOpenException e) {
88 e.printStackTrace();
89 }
90 });
91 listenerThread.start();
92 }
93 }
