231 private int[] search(String[] names, String toSearch)
232 {
233 ArrayList<Integer> arrayList = new ArrayList<Integer>();
234 for(int index = 0; index < names.length; index++)
235 {
236 if(names[index].toLowerCase().contains(toSearch.toLowerCase()))
237 {
238 arrayList.add(index);
239 }
240 }
241 return arrayList.stream().mapToInt(i -> i).toArray();
242 }
